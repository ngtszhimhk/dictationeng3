// --- 核心優化：智慧批改算法 --- async function autoGrade() { const allBoxes = document.querySelectorAll('.word-box'); let correctCount = 0; const totalCount = allBoxes.length; // 顯示「AI 批改中」的提示 const btn = document.querySelector('.btn-check'); btn.innerText = "AI 正在分析筆跡..."; btn.disabled = true; for (const box of allBoxes) { const id = box.querySelector('canvas').id.replace('canvas-', ''); const canvas = canvases[id].canvas; const targetText = document.getElementById(`ans-${id}`).innerText; // 1. 檢查是否空白 if (isCanvasBlank(canvas)) { markResult(id, false); continue; } // 2. 執行像素相似度比對 (模擬 OCR 邏輯) // 這裡透過建立一個隱形的標準答案畫布，與學生的畫布進行交集比對 const similarity = await calculateSimilarity(canvas, targetText); // 3. 設定閾值：相似度超過 40% 判定為正確 (考慮到學生字跡不標準) // 同時檢測筆劃密度，避免學生畫一個大圈就過關 if (similarity > 0.4) { markResult(id, true); correctCount++; } else { markResult(id, false); } } // 計算分數 (錯一扣三) let score = 60 - ((totalCount - correctCount) * 3); if (score < 0) score = 0; showFeedback(score, correctCount, totalCount); btn.innerText = "AI 自動批改"; btn.disabled = false; } // 模擬 OCR 的相似度計算函數 function calculateSimilarity(userCanvas, text) { return new Promise((resolve) => { const tempCanvas = document.createElement('canvas'); tempCanvas.width = userCanvas.width; tempCanvas.height = userCanvas.height; const tCtx = tempCanvas.getContext('2d'); // 在臨時畫布上寫下「標準答案」作為模板 tCtx.font = "bold 80px Arial"; // 使用標準字體 tCtx.textAlign = "center"; tCtx.textBaseline = "middle"; tCtx.fillText(text, tempCanvas.width / 2, 100); const userData = userCanvas.getContext('2d').getImageData(0,0,userCanvas.width, userCanvas.height).data; const imageData = tCtx.getImageData(0,0,tempCanvas.width, tempCanvas.height).data; let match = 0; let totalTemplatePixels = 0; for (let i = 0; i < imageData.length; i += 4) { // 如果模板這裡有字 (Alpha 通道 > 0) if (imageData[i + 3] > 0) { totalTemplatePixels++; // 如果學生也在這裡畫了線 if (userData[i + 3] > 0) { match++; } } } // 返回覆蓋率 resolve(match / totalTemplatePixels); }); }
